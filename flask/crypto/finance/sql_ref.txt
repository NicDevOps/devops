
CREATE TABLE users (
    id INTEGER, 
    username TEXT NOT NULL, 
    hash TEXT NOT NULL, 
    cash NUMERIC NOT NULL DEFAULT 10000.00, 
    PRIMARY KEY(id)
);

CREATE UNIQUE INDEX username ON users (username);

CREATE TABLE stocks (
    id INTEGER,
    user_id INTEGER,
    symbol TEXT NOT NULL,
    shares_qty INTEGER,
    price INTEGER,
    total INTEGER,
    PRIMARY KEY(id)
    FOREIGN KEY(user_id) REFERENCES users(id)
);

CREATE TABLE orders (
    id INTEGER,
    user_id INTEGER,
    order_type TEXT NOT NULL,
    symbol TEXT NOT NULL,
    shares_num INTEGER,
    share_price INTEGER,
    total INTEGER,
    date DATETIME,
    PRIMARY KEY(id),
    FOREIGN KEY(user_id) REFERENCES users(id)
);
